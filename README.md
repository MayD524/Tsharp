### Install & Run

> Install
```
$ sudo make install
```

> Run
```
$ t# <filename>.t#
```

> Hello World
```py
func main() do
    print("Hello World");
end;
```

> Function call
```py
func SampleFunc() do
    print("Function call!");
end;

func main() do
    SampleFunc();
end;
```

> Variable
```py
func main() do
    name = "T#";
    print(name);
end;
```

> Call variables from other functions
```py
func SampleFunc() do
    name = "T#";
end;

func main() do
    SampleFunc();
    print(SampleFunc.name);
end;
```

> If statement
```py
func main() do
    if 10 == 10 do
        print("Hello World!");
    end;

    if 10 != 10 do
        print("Hello World!");
    else
        print("T# Programming Language");
    end;

    if 3 < 2 do
        print("Hello World!");
    end;

    if 3 > 2 do
        print("Hello World!");
    end;

    if ("Foo" == "Foo") == ("T#" == "T#") do
        print("T# Programming Language!");
    end;
end;
```

> While loop
```py
func main() do
    num = 0;
    while num < 101 do
        print(num);
        num++;
    end;
end;
```

> FizzBuzz
```py
func main() do
    i = 1;
    while i < 101 do
        if rem(i, 15) == 0 do
            print("FizzBuzz");
        elif rem(i, 3) == 0 do
            print("Fizz");
        elif rem(i, 5) == 0 do
            print("Buzz");
        else
            print(i);
        end;
        i++;
    end;
end;
```
